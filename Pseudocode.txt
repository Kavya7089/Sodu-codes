// Reverse Number
FUNCTION reverse(n : INTEGER) : INTEGER
  DECLARE a : INTEGER
  DECLARE b : INTEGER
  SET b := 0
  WHILE n > 0 LOOP
    a := n MOD 10
    n := n / 10
    b := b * 10 + a
  END WHILE
  RETURN b
END reverse
PRINT reverse(1234)

// Factorial
FUNCTION factorial(n: INTEGER)
  DECLARE fact : INTEGER
  SET fact := 1
  FOR i in n to 1
    fact := fact * i
  END FOR
  RETURN fact
END FUNCTION
PRINT factorial(5)

// Sum the series 1 ... n
FUNCTION Sum (n : INTEGER) : INTEGER
  DECLARE s : INTEGER
  SET s := 0
  FOR i := 1 to n
    s := s + i
  END FOR
  RETURN s
END FUNCTION
PRINT Sum(5)
